{{- $img := .Page.Resources.GetMatch .Destination -}} {{- if not $img -}} {{- $img = resources.Get .Destination -}} {{- end -}} {{- if $img -}} {{- $mediaType := $img.MediaType.SubType -}} {{/* GIF, SVG, and AVIF: keep as-is */}} {{- if or (eq $mediaType "svg+xml") (eq $mediaType "svg") (eq $mediaType "gif") (eq $mediaType "avif") -}} {{/* WebP: only resize */}} {{- else if eq $mediaType "webp" -}} {{- $small := $img.Resize "800x" -}} {{- $medium := $img.Resize "1200x" -}} {{/* PNG and JPEG: convert to WebP with 90% quality and resize */}} {{- else -}} {{- $small := ($img.Resize "800x webp q90") -}} {{- $medium := ($img.Resize "1200x webp q90") -}} {{- end -}} {{- else -}} {{/* Fallback for external images */}} {{- end -}}
