{{- $img := .Page.Resources.GetMatch .Destination -}} {{- if not $img -}} {{- $img = resources.Get .Destination -}} {{- end -}} {{- if $img -}} {{- $mediaType := $img.MediaType.SubType -}} {{/* GIF and SVG: keep as-is */}} {{- if or (eq $mediaType "gif") (eq $mediaType "svg") (eq $mediaType "svg+xml") -}} {{/* WebP and AVIF: only resize (no conversion) */}} {{- else if or (eq $mediaType "webp") (eq $mediaType "avif") -}} {{- if $img.Width -}} {{- $small := $img.Resize "800x" -}} {{- $medium := $img.Resize "1200x" -}} {{- else -}} {{- end -}} {{/* PNG and JPEG: convert to WebP and resize */}} {{- else if or (eq $mediaType "png") (eq $mediaType "jpeg") (eq $mediaType "jpg") -}} {{- if $img.Width -}} {{- $small := $img.Resize "800x webp" -}} {{- $medium := $img.Resize "1200x webp" -}} {{- else -}} {{- end -}} {{/* Other formats: render as-is */}} {{- else -}} {{- end -}} {{- else -}} {{/* Fallback for external images */}} {{- end -}}
